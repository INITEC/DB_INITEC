create database initec

use initec

create table integrantes (
id_integrante int not null auto_increment primary key,
integrante varchar (200),
abreviatura varchar (5),
cargo varchar (70),
celular varchar (20),
gmail varchar (30),
e_mail varchar (30),
direccion varchar (50),
facultad varchar (10),
especialidad varchar (30),
foto varchar (50)
)

create table asistencias (
id_asistencia int not null auto_increment primary key,
hora time,
asistencia varchar (20),
condicion varchar (20),
id_integrante int,
id_fecha int
)

create table Reuniones (
id_fecha int not null auto_increment primary key,
dia_semana varchar (10),
dia int,
mes int,
ano int,
hora_inicio time,
hora_final time,
lugar varchar (200),
asunto text)
