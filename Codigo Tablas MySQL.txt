create database initec

use initec

create table integrantes (
id_integrante int not null auto_increment primary key,
integrante varchar (200),
abreviatura varchar (5),
cargo varchar (70),
celular varchar (20),
gmail varchar (50),
e_mail varchar (50),
direccion varchar (50),
facultad varchar (10),
especialidad varchar (30),
foto varchar (50),
usuario varchar(50),
clave varchar(50),
trabajo varchar (50)
)

create table asistencias (
id_asistencia int not null auto_increment primary key,
hora time,
asistencia varchar (20),
condicion varchar (20),
id_integrante int,
id_fecha int
)

create table reuniones (
id_fecha int not null auto_increment primary key,
dia_semana varchar (10),
hora_inicio time,
hora_final time,
lugar varchar (200),
asunto text,
fecha date)

create table examenes (
id_examen int not null auto_increment primary key,
examen varchar (200),
fecha date,
n_maxima int,
n_aprobatoria int)

create table notas(
id_nota int not null auto_increment primary key,
nota int,
condicion varchar(50),
id_integrante int,
id_examen int)

create table amonestaciones (
id_amonestacion int not null auto_increment primary key,
receptor int,
remitente int,
motivo varchar(200),
fecha_emision date,
fecha_falta date,
tipo varchar(20)
)

create table justificaciones (
id_justificacion int not null auto_increment primary key,
id_integrante int,
id_fecha int,
condicion_justificacion varchar(20),
asistencia_justificacion varchar(20),
motivo text,
fecha_justificacion date,
hora_justificacion time
)

create table trabajos (
id_trabajo int not null auto_increment primary key,
trabajo varchar(50),
tarea varchar(50),
grafico varchar(50),
direccion varchar(100)
)
